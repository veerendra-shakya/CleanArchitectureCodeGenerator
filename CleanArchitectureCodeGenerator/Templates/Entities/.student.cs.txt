using System.Collections.Generic;
using {selectns}.Common.Entities;

{

    public string ContactNumber { get; set; }

    // One-to-One Relationship with DemoProfile
    [Scaffolding(PropRole.Relationship, RelationshipType.OneToOne, inverseProperty: "Student",  foreignKeyProperty: "StudentId",  DeleteBehavior.Cascade)]

    [Scaffolding(PropRole.Relationship, RelationshipType.ManyToOne, inverseProperty: "Students", foreignKeyProperty: "SchoolId", DeleteBehavior.Restrict)]

    // Many-to-Many Relationship with DemoCourse
    [Scaffolding(PropRole.Relationship, RelationshipType.ManyToMany, inverseProperty: "Students")]
